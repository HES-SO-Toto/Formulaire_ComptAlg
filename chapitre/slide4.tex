  \formdesc{Intégration de Gauss}

  $$\int_{a}^{b} f(x) dx = \sum_{i=1}^{n} w_i f(x_i)$$

  $$w_i = \int_{a}^{b} \prod_{j=0, j \neq i}^{n} \cfrac{x-x_j}{x_i-x_j} dx = \int_{a}^{b} \prod_{j=0, j \neq i}^{n} \cfrac{x-x_j}{x_i-x_j} dx$$

\formtitle{Polynomes de Legendre :}

$P_n(x) = \cfrac{1}{2^n n!} \cfrac{d^n}{dx^n} \left[ (x^2 - 1)^n \right]$

$P_0(x) = 1 \quad P_1(x) = x \quad P_2(x) = \cfrac{1}{2} (3x^2 - 1)$

$P_3(x) = \cfrac{1}{2} (5x^3 - 3x) \quad P_4(x) = \cfrac{1}{8} (35x^4 - 30x^2 + 3)$
